csv :      enable/disable csv mod.
Arguments:
 -e - enable
 -d - disable

pwr :      on/off controller pin responsible
           for the power supply.
Arguments:
--on 
--off

setlvl :   set alert levels for one tmp. sensor
           in format - 
           <number of sensor>,<low level>,<highlevel> 
Arguments:
 -p - print set value.

shwparam : show parameters of all, selected range
           or single tmp sensors.
           Format of ranges for example -
           0-5,7-9,10,11-20
Arguments:
--all - show parameters of all tmp sensors(32)

tsend :    enable/disable sending of temperature
           measurements to the PC.
Arguments:
 -e - enable
 -d - disable

echo :     on/off command echo
Arguments:
--on 
--off

setrng :   set alert levels for range of tmp sensors
           or single tmp sensor.
           Format of ranges for example -
           0-5,7-9,10,11-20
Arguments:
 -p - print set value.

tchck : NOT UTILIZED

tinit : NOT UTILIZED

disp :     Set number of display tmp sensors in
           one string
Arguments:
--all - display all tmp sensors measurements
